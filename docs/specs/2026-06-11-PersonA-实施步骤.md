## Step 2: RKLLM 模型加载与推理封装（1天）

### 2.1 创建模型引擎 `src/ai/rkllm_engine.py`

```python
"""RKLLM 模型加载与推理封装"""
import time
from typing import Optional


class RKLLMEngine:
    """
    DeepSeek-R1-Distill-Qwen-1.5B 模型的 RKLLM 封装。

    使用示例:
        engine = RKLLMEngine("models/DeepSeek-R1-Distill-Qwen-1.5B_W8A8_RK3588.rkllm")
        engine.init()
        response = engine.chat("你好")
        engine.shutdown()
    """

    def __init__(self, model_path: str):
        self.model_path = model_path
        self.llm = None

    def init(self) -> bool:
        """加载模型到 NPU"""
        try:
            from rkllm import LLM
            print(f"[RKLLM] 加载模型: {self.model_path}")
            self.llm = LLM(self.model_path)
            self.llm.init()
            print("[RKLLM] 模型加载完成")
            return True
        except ImportError:
            print("[RKLLM] rkllm 库未安装，将使用降级模式（本地规则匹配）")
            self.llm = None
            return True  # 返回 True 允许系统在无模型时降级运行
        except Exception as e:
            print(f"[RKLLM] 模型加载失败: {e}")
            self.llm = None
            return False

    def generate(self, prompt: str, max_tokens: int = 256,
                 temperature: float = 0.1, top_p: float = 0.9) -> str:
        """
        调用 LLM 推理。

        Args:
            prompt: 完整的 prompt 文本
            max_tokens: 最大生成 token 数
            temperature: 温度参数（器件匹配任务建议 0.1，保证输出稳定）
            top_p: nucleus sampling 参数

        Returns:
            LLM 生成的文本
        """
        if self.llm is None:
            return self._fallback_generate(prompt)

        try:
            t0 = time.time()
            # RKLLM 推理调用（具体 API 以实际 SDK 为准）
            response = self.llm.generate(
                prompt,
                max_new_tokens=max_tokens,
                temperature=temperature,
                top_p=top_p,
            )
            elapsed = time.time() - t0
            print(f"[RKLLM] 推理耗时 {elapsed:.2f}s, 输出 {len(response)} 字符")
            return response
        except Exception as e:
            print(f"[RKLLM] 推理失败: {e}")
            return self._fallback_generate(prompt)

    def _fallback_generate(self, prompt: str) -> str:
        """
        降级模式：当模型不可用时，返回提示。
        实际场景中永远不应走到这里（模型必然存在），
        但保证 Person C 在无 NPU 的 PC 上也能用 Mock 开发。
        """
        return '{"component_id": null, "component_name": "", "cabinet_id": null, "confidence": 0.0, "response": "AI 模型未加载，请检查 RKLLM 环境"}'

    def is_available(self) -> bool:
        """模型是否可用"""
        return self.llm is not None

    def shutdown(self):
        """释放模型资源"""
        if self.llm is not None:
            # 根据实际 SDK API 调用释放方法
            print("[RKLLM] 模型资源已释放")
```

```python
"""Prompt 模板管理 — 器件匹配场景专用"""


# ── 系统指令（每次查询时拼入 prompt 开头） ──
SYSTEM_INSTRUCTION = """你是一个实验室器件管理助手。用户会用自然语言描述需要的电子器件，
你需要从器件数据库中找出最匹配的器件并回复。

规则：
1. 如果找到匹配器件，confidence 设为 0.7-1.0
2. 如果没有精确匹配但有关联器件，confidence 设为 0.3-0.6，放在 alternatives 中
3. 如果完全找不到，component_id 和 cabinet_id 设为 null，confidence 为 0
4. response 用中文自然语言回复，告诉用户找到了什么、在哪个柜子
5. 只输出 JSON，不要输出任何其他文字

输出 JSON 格式（严格遵守）：
{"component_id": 数字或null, "component_name": "器件名", "cabinet_id": 数字或null, "confidence": 0.0-1.0, "response": "给用户的回复", "alternatives": []}"""


def build_component_catalog(components_summary: str) -> str:
    """构建器件目录描述"""
    return f"""当前器件数据库：

{components_summary}

注意：所有器件都在数据库中，柜门编号即 cabinet_id。
如果用户的需求电压和某个器件电压接近（±0.5V），优先推荐。
如果用户用简称（如"51单片机"→STC89C51），尝试智能匹配。"""


def build_query_prompt(user_query: str, component_catalog: str) -> str:
    """
    构建完整的查询 prompt。

    Args:
        user_query: 用户的原始语音识别文本，如"我需要一个5V的稳压芯片"
        component_catalog: 通过 DataManager.get_all_components_for_llm() 获得的器件目录

    Returns:
        完整的 prompt 字符串
    """
    catalog_section = build_component_catalog(component_catalog)

    return f"""{SYSTEM_INSTRUCTION}

{catalog_section}

用户说：「{user_query}」

请根据用户需求匹配最合适的器件（JSON格式）："""


def build_conversation_prompt(user_query: str, component_catalog: str,
                               history: list = None) -> str:
    """
    构建带对话历史的 prompt。

    Args:
        user_query: 当前用户输入
        component_catalog: 器件目录
        history: 对话历史 [{"role": "user/assistant", "content": "..."}]
    """
    # 对 1.5B 模型，历史不宜超过 2 轮
    history_text = ""
    if history:
        recent = history[-4:]  # 只保留最近 2 轮（4 条）
        for h in recent:
            role = "用户" if h["role"] == "user" else "助手"
            history_text += f"{role}: {h['content']}\n"

    catalog_section = build_component_catalog(component_catalog)

    return f"""{SYSTEM_INSTRUCTION}

{catalog_section}

{history_text}
用户: {user_query}
助手: """


# ── Few-shot 示例（协助小模型稳定输出） ──

FEW_SHOT_EXAMPLES = """
示例1:
用户说：「找一个五伏的电源芯片」
输出: {"component_id": 1, "component_name": "LM7805", "cabinet_id": 1, "confidence": 0.95, "response": "为您找到 LM7805，5V 线性稳压芯片，存放在 1 号柜。", "alternatives": []}

示例2:
用户说：「来个能连WiFi的模块」
输出: {"component_id": 5, "component_name": "ESP32-WROOM-32", "cabinet_id": 5, "confidence": 0.90, "response": "为您找到 ESP32-WROOM-32，WiFi 蓝牙双模模块，存放在 5 号柜。", "alternatives": []}

示例3:
用户说：「有没有51单片机」
输出: {"component_id": null, "component_name": "", "cabinet_id": null, "confidence": 0.0, "response": "抱歉，目前库存中没有 51 单片机，我们有 STM32F103C8T6（ARM Cortex-M3），需要看看吗？", "alternatives": [{"id": 4, "name": "STM32F103C8T6", "cabinet_id": 4}]}
"""


def build_query_prompt_with_examples(user_query: str, component_catalog: str) -> str:
    """
    带 few-shot 示例的查询 prompt（推荐使用，输出更稳定）。
    """
    catalog_section = build_component_catalog(component_catalog)

    return f"""{SYSTEM_INSTRUCTION}

{catalog_section}

{FEW_SHOT_EXAMPLES}

现在请处理:
用户说：「{user_query}」
输出: """
```

### 4.1 `src/ai/response_parser.py`

```python
"""LLM 输出解析器 — 从 LLM 返回的文本中提取结构化的 MatchResult"""
import json
import re
from typing import Optional
from src.interfaces import MatchResult


def parse_llm_response(raw_text: str) -> MatchResult:
    """
    从 LLM 返回的原始文本中提取 MatchResult。

    容错策略:
    1. 尝试直接解析 JSON
    2. 尝试提取 ```json ... ``` 代码块
    3. 尝试正则提取 JSON 对象
    4. 全部失败则返回空结果

    Args:
        raw_text: LLM 生成的原始文本

    Returns:
        MatchResult 对象
    """
    # 策略1: 直接解析
    result = _try_parse_json(raw_text)
    if result:
        return result

    # 策略2: 提取 markdown 代码块
    code_block = re.search(r'```(?:json)?\s*\n?(.*?)\n?```', raw_text, re.DOTALL)
    if code_block:
        result = _try_parse_json(code_block.group(1))
        if result:
            return result

    # 策略3: 正则提取最外层的 JSON 对象
    json_obj = re.search(r'\{[^{}]*\}', raw_text)
    if json_obj:
        result = _try_parse_json(json_obj.group(0))
        if result:
            return result

    # 策略4: 全部失败，返回空结果
    return MatchResult(
        component_id=None,
        component_name="",
        cabinet_id=None,
        confidence=0.0,
        response_text=raw_text.strip()[:200],  # 截取前 200 字符作为回复
        alternatives=[],
    )


def _try_parse_json(text: str) -> Optional[MatchResult]:
    """尝试将文本解析为 MatchResult"""
    try:
        data = json.loads(text.strip())
        return MatchResult(
            component_id=data.get("component_id"),
            component_name=data.get("component_name", ""),
            cabinet_id=data.get("cabinet_id"),
            confidence=float(data.get("confidence", 0.0)),
            response_text=data.get("response", ""),
            alternatives=data.get("alternatives", []),
        )
    except (json.JSONDecodeError, ValueError, TypeError):
        return None
```

### 5.1 `src/ai/ai_engine.py`

```python
"""AIEngine 实现 — 对接 src/interfaces.py 中的 AIEngine 接口"""
import time
from typing import Dict, Optional

from src.interfaces import AIEngine, MatchResult
from src.ai.rkllm_engine import RKLLMEngine
from src.ai.prompt_templates import build_query_prompt_with_examples
from src.ai.response_parser import parse_llm_response


class AIEngineImpl(AIEngine):
    """
    AIEngine 的完整实现。

    依赖:
    - RKLLMEngine: 模型推理
    - DataManager.get_all_components_for_llm(): 器件目录（由 Person C 提供）

    使用:
        engine = AIEngineImpl()
        engine.initialize()
        engine.set_data_manager(data_manager)   # Person C 注入
        result = engine.process_query("我需要5V稳压芯片")
        print(result.response_text)  # "为您找到 LM7805，存放在 1 号柜"
    """

    def __init__(self):
        self.rkllm: Optional[RKLLMEngine] = None
        self.data_manager = None      # Person C 注入
        self._catalog_cache: str = ""
        self._catalog_cache_time: float = 0
        self._cache_ttl: float = 60.0  # 器件目录缓存 60 秒

    def initialize(self, model_path: str = "") -> bool:
        """加载模型"""
        if not model_path:
            model_path = "models/DeepSeek-R1-Distill-Qwen-1.5B_W8A8_RK3588.rkllm"

        self.rkllm = RKLLMEngine(model_path)
        return self.rkllm.init()

    def set_data_manager(self, dm):
        """
        注入 DataManager（由 Person C 在系统初始化时调用）。
        这样 AIEngine 就可以获取实时器件目录。
        """
        self.data_manager = dm

    def process_query(self, query_text: str,
                       context: Optional[Dict] = None) -> MatchResult:
        """
        处理自然语言查询的主入口。

        流程:
        1. 获取器件目录（优先从 DataManager，否则用缓存）
        2. 构建 prompt（含 few-shot 示例）
        3. LLM 推理
        4. 解析 JSON → MatchResult
        5. 如果 LLM 不可用，降级到关键词匹配
        """
        query_text = query_text.strip()
        if not query_text:
            return MatchResult(
                component_id=None, component_name="", cabinet_id=None,
                confidence=0.0, response_text="请说出您需要的器件", alternatives=[]
            )

        # 获取器件目录
        catalog = self._get_catalog()

        # LLM 推理
        if self.rkllm and self.rkllm.is_available():
            return self._llm_match(query_text, catalog, context)
        else:
            return self._fallback_match(query_text)

    def update_catalog(self, catalog_text: str = ""):
        """强制刷新器件目录缓存"""
        if catalog_text:
            self._catalog_cache = catalog_text
        elif self.data_manager:
            self._catalog_cache = self.data_manager.get_all_components_for_llm()
        self._catalog_cache_time = time.time()

    def _get_catalog(self) -> str:
        """获取器件目录（带缓存）"""
        now = time.time()
        if (not self._catalog_cache or
                now - self._catalog_cache_time > self._cache_ttl):
            if self.data_manager:
                self._catalog_cache = self.data_manager.get_all_components_for_llm()
            else:
                self._catalog_cache = "（器件数据库暂未连接）"
            self._catalog_cache_time = now
        return self._catalog_cache

    def _llm_match(self, query: str, catalog: str,
                    context: Optional[Dict]) -> MatchResult:
        """LLM 语义匹配"""
        try:
            prompt = build_query_prompt_with_examples(query, catalog)
            raw_response = self.rkllm.generate(prompt, max_tokens=256, temperature=0.1)
            result = parse_llm_response(raw_response)
            return result
        except Exception as e:
            print(f"[AIEngine] LLM 匹配异常: {e}，降级到关键词匹配")
            return self._fallback_match(query)

    def _fallback_match(self, query: str) -> MatchResult:
        """
        降级匹配（当 LLM 不可用时）。
        使用 src/voice/nlu_processor.py 中已有的关键词匹配逻辑。
        """
        try:
            from src.voice.nlu_processor import NLUProcessor
            nlu = NLUProcessor("data/components.json")
            result = nlu.process_query(query)
            if result["has_match"]:
                best = result["best_match"]
                comp = best["component"]
                return MatchResult(
                    component_id=comp.get("id"),
                    component_name=comp["name"],
                    cabinet_id=comp.get("cabinet_id"),
                    confidence=best["score"],
                    response_text=f"为您找到 {comp['name']}，存放在 {comp.get('cabinet_id', '未知')} 号柜",
                )
        except Exception as e:
            print(f"[AIEngine] 降级匹配也失败: {e}")

        return MatchResult(
            component_id=None, component_name="", cabinet_id=None,
            confidence=0.0,
            response_text="抱歉，AI 服务暂时不可用，请联系管理员",
        )

    def shutdown(self):
        if self.rkllm:
            self.rkllm.shutdown()
```

### 6.2 语音识别增强

现有 `src/voice/voice_recognizer.py` 已经有基础框架。需要增加：

修改 `voice_recognizer.py`，在现有 `_process_audio_thread` 中添加 VAD：

```python
# 在 VoiceRecognizer 类中添加

def __init__(self, ...):
    ...
    self.speech_threshold = 500       # 音量阈值
    self.silence_timeout = 1.5        # 静音 1.5 秒视为说话结束
    self.wake_word = "你好储物柜"     # 可选的唤醒词

def _process_audio_thread(self):
    """音频处理线程 — 增加 VAD"""
    import struct

    speaking = False
    silence_start = None
    audio_buffer = []

    while self.is_listening:
        try:
            data = self.audio_queue.get(timeout=0.1)

            # 简单音量检测
            audio_data = struct.unpack(f'<{len(data)//2}h', data)
            volume = max(abs(x) for x in audio_data) if audio_data else 0

            if volume > self.speech_threshold:
                if not speaking:
                    speaking = True
                    audio_buffer = []
                    print("[VAD] 检测到语音...")
                audio_buffer.append(data)
                silence_start = None
            elif speaking:
                audio_buffer.append(data)
                if silence_start is None:
                    silence_start = time.time()
                elif time.time() - silence_start > self.silence_timeout:
                    # 说话结束，处理完整语音
                    speaking = False
                    print(f"[VAD] 语音结束，缓冲区 {len(audio_buffer)} 帧")
                    self._process_complete_utterance(b''.join(audio_buffer))
                    audio_buffer = []

            # 原有的逐帧识别（实时反馈）
            if self.recognizer.AcceptWaveform(data):
                result = self.recognizer.Result()
                if self.callback:
                    self.callback(result)

        except queue.Empty:
            continue

def _process_complete_utterance(self, audio_data: bytes):
    """处理完整的语音段"""
    # 创建临时识别器（避免与实时识别器冲突）
    from vosk import KaldiRecognizer
    temp_rec = KaldiRecognizer(self.model, self.sample_rate)
    if temp_rec.AcceptWaveform(audio_data):
        import json
        result = json.loads(temp_rec.Result())
        text = result.get("text", "").strip()
        if text and self.callback:
            self.callback(json.dumps(result))
```

> **注意**: 上述 VAD 是最简方案，仅基于音量。如果效果不好，可用 `webrtcvad` 库替代（`pip install webrtcvad`）。

### 6.3 语音交互状态机 `src/voice/interaction_manager.py`

```python
"""语音交互状态机 — 管理 监听→识别→处理→播报 的完整流程"""
import threading
import time
import queue
from enum import Enum, auto


class InteractionState(Enum):
    IDLE = auto()          # 空闲，等待唤醒
    LISTENING = auto()     # 正在听
    PROCESSING = auto()    # AI 处理中
    SPEAKING = auto()      # 正在播报


class InteractionManager:
    """
    语音交互状态机。

    使用:
        manager = InteractionManager(ai_engine, voice_recognizer, voice_synthesizer)
        manager.on_result = lambda result: print(result.response_text)
        manager.start()

    流程:
        IDLE → (语音检测) → LISTENING → (VAD结束) → PROCESSING
        → (LLM返回) → SPEAKING → (播报完成) → IDLE
    """

    def __init__(self, ai_engine, voice_recognizer, voice_synthesizer):
        self.ai_engine = ai_engine
        self.recognizer = voice_recognizer
        self.synthesizer = voice_synthesizer

        self.state = InteractionState.IDLE
        self._running = False
        self._result_callback = None

        # 超时设置
        self.idle_timeout = 60.0       # 60 秒无交互→播报提示
        self.listening_timeout = 15.0  # 15 秒没说话→回到 IDLE
        self._last_activity = time.time()

    def start(self):
        """启动状态机"""
        self._running = True
        self.recognizer.start_listening(self._on_speech_result)
        threading.Thread(target=self._state_loop, daemon=True).start()
        print("[Interaction] 语音交互已启动 (唤醒词: '你好储物柜')")

    def stop(self):
        """停止"""
        self._running = False
        self.recognizer.stop_listening()

    def _on_speech_result(self, result_json: str):
        """语音识别回调"""
        import json
        text = json.loads(result_json).get("text", "").strip()
        if not text:
            return

        print(f"[Interaction] 识别: {text}")
        self._last_activity = time.time()

        if self.state == InteractionState.IDLE:
            # 检查唤醒词
            if "你好储物柜" in text or "储物柜" in text:
                self.state = InteractionState.LISTENING
                self.synthesizer.speak_async("在，请说")
            return

        if self.state == InteractionState.LISTENING:
            self.state = InteractionState.PROCESSING
            self._process_query(text)

    def _process_query(self, text: str):
        """AI 处理 → 播报"""
        # 在独立线程中处理，避免阻塞音频回调
        def worker():
            result = self.ai_engine.process_query(text)
            self.state = InteractionState.SPEAKING
            self.synthesizer.speak(result.response_text, blocking=True)
            self.state = InteractionState.IDLE
            self._last_activity = time.time()

            if self._result_callback:
                self._result_callback(result)

        threading.Thread(target=worker, daemon=True).start()

    def _state_loop(self):
        """状态监控循环（超时处理）"""
        while self._running:
            time.sleep(5)
            idle_time = time.time() - self._last_activity

            if (self.state == InteractionState.IDLE and
                    idle_time > self.idle_timeout):
                self.synthesizer.speak_async("系统等待您的指令")
                self._last_activity = time.time()

            if (self.state == InteractionState.LISTENING and
                    idle_time > self.listening_timeout):
                self.state = InteractionState.IDLE
                print("[Interaction] 监听超时，回到 IDLE")
```

## 常见问题

**Q: rkllm 库找不到？**
```bash
# 检查 Rockchip SDK 是否已安装
find / -name "rkllm*" -o -name "librkllm*" 2>/dev/null
# 如果在 /usr/local/lib/ 下，确保 Python 能找到
export LD_LIBRARY_PATH=/usr/local/lib:$LD_LIBRARY_PATH
```

**Q: 模型推理很慢（>10秒）？**
- 检查 NPU 驱动是否加载: `dmesg | grep -i npu`
- 确认模型确实是 W8A8 量化（非 FP16）
- 减少 `max_tokens`（器件匹配场景 128-256 足够）

**Q: LLM 输出的 JSON 格式不稳定？**
- 使用 `build_query_prompt_with_examples()`（含 few-shot）
- 降低 temperature 到 0.05-0.1
- 增加 `response_parser.py` 的容错策略

**Q: pyaudio 在 ARM 上装不上？**
```bash
sudo apt install portaudio19-dev
pip install pyaudio
# 或者用 sounddevice 替代
pip install sounddevice
# 修改 voice_recognizer.py 使用 sounddevice 替代 pyaudio
```
