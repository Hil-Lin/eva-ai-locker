## Step 5: 理解 C API（为 Python 封装做准备）

### 5.1 看 Demo 源码的关键部分

`llm_demo.cpp` 中的核心流程：

```c
// ① 初始化参数
RKLLMParam param = rkllm_createDefaultParam();
param.model_path = "模型路径.rkllm";

// ② 初始化模型
rkllm_init(&llmHandle, &param, callback);

// ③ 构造输入（注意特殊格式！）
text = "<｜begin▁of▁sentence｜><｜User｜>" + 用户输入 + "<｜Assistant｜>";

// ④ 推理
RKLLMInput rkllm_input;
rkllm_input.input_type = RKLLM_INPUT_PROMPT;
rkllm_input.prompt_input = text;
rkllm_run(llmHandle, &rkllm_input, &rkllm_infer_params, NULL);

// ⑤ 结果通过 callback 异步返回
void callback(RKLLMResult *result, void *userdata, LLMCallState state) {
    if (state == RKLLM_RUN_NORMAL) {
        printf("%s", result->text);  // 逐个 token 输出
    }
}
```

### 5.2 prompt 格式（非常重要！）

```
<｜begin▁of▁sentence｜><｜User｜>你的问题<｜Assistant｜>
```

这是 DeepSeek-R1 模型要求的 chat template。**必须按这个格式拼接 prompt**。

## Step 6: 写 Python ctypes 封装（你的核心工作）

> 因为 F 盘里的 RKLLM SDK **没有 ARM64 的 Python wheel 包**（只有 C 运行时 `librkllmrt.so`），
> 所以你需要用 Python 的 `ctypes` 库来调用 C API。

### 6.1 `src/ai/rkllm_engine.py`

```python
"""
RKLLM Python 封装 — 通过 ctypes 调用 librkllmrt.so
"""
import ctypes
import os
import time
import threading
from typing import Optional, Callable

# ── C 结构体定义（对照 rkllm.h） ──

class RKLLMParam(ctypes.Structure):
    _fields_ = [
        ("model_path", ctypes.c_char_p),
        ("top_k", ctypes.c_int),
        ("top_p", ctypes.c_float),
        ("temperature", ctypes.c_float),
        ("repeat_penalty", ctypes.c_float),
        ("frequency_penalty", ctypes.c_float),
        ("presence_penalty", ctypes.c_float),
        ("max_new_tokens", ctypes.c_int),
        ("max_context_len", ctypes.c_int),
        ("skip_special_token", ctypes.c_bool),
    ]

class RKLLMResult(ctypes.Structure):
    _fields_ = [("text", ctypes.c_char_p)]

class RKLLMInput(ctypes.Structure):
    _fields_ = [
        ("input_type", ctypes.c_int),
        ("prompt_input", ctypes.c_char_p),
    ]

class RKLLMInferParam(ctypes.Structure):
    _fields_ = [("mode", ctypes.c_int)]

# ── 回调函数类型 ──
CALLBACK_TYPE = ctypes.CFUNCTYPE(None, ctypes.POINTER(RKLLMResult),
                                   ctypes.c_void_p, ctypes.c_int)

# ── Python 封装 ──

LLM_HANDLE = ctypes.c_void_p
RKLLM_INPUT_PROMPT = 0
RKLLM_INFER_GENERATE = 0
RKLLM_RUN_NORMAL = 0
RKLLM_RUN_FINISH = 1
RKLLM_RUN_ERROR = 2

# prompt 模板
PROMPT_PREFIX = b"<|\x62\x65\x67\x69\x6e\x5f\x6f\x66\x5f\x73\x65\x6e\x74\x65\x6e\x63\x65|><｜User｜>"
PROMPT_POSTFIX = b"<｜Assistant｜>"
# 以上是 "<｜begin▁of▁sentence｜><｜User｜>" 和 "<｜Assistant｜>" 的字节表示


def _create_default_param(model_path: str) -> RKLLMParam:
    """创建默认参数（参照 llm_demo.cpp）"""
    param = RKLLMParam()
    param.model_path = model_path.encode('utf-8')
    param.top_k = 1
    param.top_p = 0.95
    param.temperature = 0.1  # 器件匹配用低温，保证输出稳定
    param.repeat_penalty = 1.1
    param.frequency_penalty = 0.0
    param.presence_penalty = 0.0
    param.max_new_tokens = 256
    param.max_context_len = 4096
    param.skip_special_token = True
    return param


class RKLLMEngine:
    """
    DeepSeek-R1-Distill-Qwen-1.5B RK3588 NPU 推理引擎。
    通过 ctypes 封装 librkllmrt.so 的 C API。
    """

    def __init__(self, model_path: str, lib_path: str = None):
        self.model_path = model_path
        self._handle = None
        self._lib = None
        self._callback_ref = None
        self._response_buffer = ""
        self._response_complete = threading.Event()

        # 找动态库
        if lib_path is None:
            lib_path = self._find_lib()
        self.lib_path = lib_path

    def _find_lib(self) -> str:
        """自动查找 librkllmrt.so"""
        candidates = [
            "lib/librkllmrt.so",
            "/opt/smart-locker/demo/lib/librkllmrt.so",
            "/usr/lib/librkllmrt.so",
            "/usr/local/lib/librkllmrt.so",
        ]
        for path in candidates:
            if os.path.exists(path):
                return path

        # 尝试全局搜索
        import subprocess
        result = subprocess.run(
            ["find", "/", "-name", "librkllmrt.so", "-type", "f"],
            capture_output=True, text=True, timeout=5
        )
        if result.stdout.strip():
            return result.stdout.strip().split('\n')[0]

        raise FileNotFoundError("找不到 librkllmrt.so")

    def init(self) -> bool:
        """加载模型到 NPU"""
        try:
            self._lib = ctypes.CDLL(self.lib_path)

            # 定义函数签名
            self._lib.rkllm_createDefaultParam.restype = RKLLMParam

            self._lib.rkllm_init.argtypes = [
                ctypes.POINTER(LLM_HANDLE),
                ctypes.POINTER(RKLLMParam),
                CALLBACK_TYPE,
            ]
            self._lib.rkllm_init.restype = ctypes.c_int

            self._lib.rkllm_run.argtypes = [
                LLM_HANDLE,
                ctypes.POINTER(RKLLMInput),
                ctypes.POINTER(RKLLMInferParam),
                ctypes.c_void_p,
            ]
            self._lib.rkllm_run.restype = ctypes.c_int

            self._lib.rkllm_destroy.argtypes = [LLM_HANDLE]
            self._lib.rkllm_destroy.restype = ctypes.c_int

            # 创建参数
            param = _create_default_param(self.model_path)

            # 创建回调
            self._callback_ref = CALLBACK_TYPE(self._stream_callback)

            # 初始化
            handle = LLM_HANDLE()
            ret = self._lib.rkllm_init(
                ctypes.byref(handle),
                ctypes.byref(param),
                self._callback_ref,
            )

            if ret != 0:
                print(f"[RKLLM] 初始化失败，返回码: {ret}")
                return False

            self._handle = handle
            print(f"[RKLLM] 模型加载成功: {self.model_path}")
            return True

        except Exception as e:
            print(f"[RKLLM] 加载异常: {e}")
            return False

    def _stream_callback(self, result_ptr, userdata, state):
        """C 回调 — 收集推理输出"""
        if state == RKLLM_RUN_NORMAL:
            if result_ptr and result_ptr.contents.text:
                text = result_ptr.contents.text.decode('utf-8', errors='replace')
                self._response_buffer += text
        elif state == RKLLM_RUN_FINISH:
            self._response_complete.set()
        elif state == RKLLM_RUN_ERROR:
            print("[RKLLM] 推理错误")
            self._response_complete.set()

    def generate(self, prompt: str, max_tokens: int = 256) -> str:
        """
        调用 LLM 推理。

        Args:
            prompt: 完整的 prompt 文本（不含 chat template 标记，会自动添加）
            max_tokens: 最大生成 token 数

        Returns:
            LLM 生成的文本
        """
        if self._handle is None:
            return '{"error": "模型未加载"}'

        # 拼接 chat template
        full_prompt = PROMPT_PREFIX + prompt.encode('utf-8') + PROMPT_POSTFIX

        # 构造输入
        rkllm_input = RKLLMInput()
        rkllm_input.input_type = RKLLM_INPUT_PROMPT
        rkllm_input.prompt_input = full_prompt

        # 推理参数
        infer_param = RKLLMInferParam()
        infer_param.mode = RKLLM_INFER_GENERATE

        # 重置缓冲区
        self._response_buffer = ""
        self._response_complete.clear()

        # 调用推理（同步阻塞直到 callback 收到 FINISH）
        ret = self._lib.rkllm_run(
            self._handle,
            ctypes.byref(rkllm_input),
            ctypes.byref(infer_param),
            None,
        )

        # 等待完成
        self._response_complete.wait(timeout=30)

        return self._response_buffer

    def is_available(self) -> bool:
        return self._handle is not None

    def shutdown(self):
        if self._handle is not None and self._lib is not None:
            self._lib.rkllm_destroy(self._handle)
            self._handle = None
            print("[RKLLM] 模型已释放")


# ── 测试 ──
if __name__ == "__main__":
    engine = RKLLMEngine("/opt/smart-locker/models/DeepSeek-R1-Distill-Qwen-1.5B_W8A8_RK3588.rkllm")

    if not engine.init():
        print("模型加载失败")
        exit(1)

    response = engine.generate("你好，请用一句话介绍你自己。", max_tokens=128)
    print(f"回复: {response}")

    engine.shutdown()
```

```python
"""AIEngine 实现 — 实现 src/interfaces.py 的 AIEngine 接口"""
from typing import Dict, Optional
from src.interfaces import AIEngine, MatchResult
from src.ai.rkllm_engine import RKLLMEngine

from src.ai.response_parser import parse_llm_response


class AIEngineImpl(AIEngine):

    def __init__(self):
        self._rkllm: Optional[RKLLMEngine] = None
        self._data_manager = None
        self._catalog = ""

    def initialize(self, model_path: str = "") -> bool:
        if not model_path:
            model_path = "/opt/smart-locker/models/DeepSeek-R1-Distill-Qwen-1.5B_W8A8_RK3588.rkllm"
        self._rkllm = RKLLMEngine(model_path)
        return self._rkllm.init()

    def set_data_manager(self, dm):
        """Person C 注入 DataManager"""
        self._data_manager = dm

    def process_query(self, query_text: str, context: Dict = None) -> MatchResult:
        if not query_text.strip():
            return MatchResult(component_id=None, component_name="", cabinet_id=None,
                               confidence=0.0, response_text="请说出您需要的器件")

        # 更新器件目录
        self._catalog = self._data_manager.get_all_components_for_llm() if self._data_manager else ""

        # 构造 prompt 并推理
        prompt = build_prompt(query_text, self._catalog)
        raw = self._rkllm.generate(prompt, max_tokens=256)
        return parse_llm_response(raw)

    def update_catalog(self, catalog_text: str = ""):
        self._catalog = catalog_text

    def shutdown(self):
        if self._rkllm:
            self._rkllm.shutdown()
```
