## Step 1: PWR.A153 B 接线与 GPIO 验证（0.5天）

### 1.1 扩展板安装

```
PWR.A153 B 扩展板直接插入 RK3588 的 40pin GPIO 排针。
对齐方向：扩展板的排母对准 RK3588 的排针，确保缺口方向一致。

        扩展板丝印布局:
        ┌──────────────────────────┐
        │   [M1+] [M1-] [M2+] [M2-]│ ← L298N电机端子（本项目不用）
        │                          │
        │   [DC电源插座]            │ ← 插 5V/3A 电源
        │                          │
        │   ┌──┬──┬──┬──┬──┬──┬──┐│
        │   │1 │2 │3 │4 │5 │6 │7 │8 ││ ← 8路舵机接口
        │   │黄│黄│黄│黄│黄│黄│黄│黄││   黄=PWM信号
        │   │红│红│红│红│红│红│红│红││   红=5V
        │   │黑│黑│黑│黑│黑│黑│黑│黑││   黑=GND
        │   └──┴──┴──┴──┴──┴──┴──┘│
        │                          │
        │       [RK3588 40pin]     │ ← 排母，插到RK3588
        └──────────────────────────┘
```

### 1.2 接线步骤

```bash
# 1. RK3588 断电
# 2. PWR.A153 B 排母对准 RK3588 40pin 排针，轻轻压紧
# 3. 舵机线按颜色插入第1路：
#    舵机棕色线 → 扩展板黑色排针 (GND)
#    舵机红色线 → 扩展板红色排针 (5V)
#    舵机橙色线 → 扩展板黄色排针 (PWM信号)
# 4. DC 插座插入 5V/3A 电源适配器
# 5. RK3588 上电
```

### 1.4 确定舵机对应的 GPIO 引脚

**PWR.A153 B 的 8 路舵机各接一个 GPIO 引脚**。你需要确定每个通道对应哪个 GPIO。

**方法：用万用表蜂鸣档**
1. 万用表打到蜂鸣档
2. 一支表笔接触舵机接口黄色排针（第1路）
3. 另一支表笔逐个接触 40pin 排针
4. 蜂鸣响的那根就是第1路对应的 GPIO 引脚

**如果没万用表，用以下引脚作为起点尝试**（树莓派兼容板的常见映射）：

| 舵机通道 | 推测 GPIO（BCM编号） | 推测 GPIO 芯片与线号 |
|----------|---------------------|---------------------|
| 通道 1 | GPIO12 | gpiochip0:12 |
| 通道 2 | GPIO13 | gpiochip0:13 |
| 通道 3 | GPIO18 | gpiochip0:18 |
| 通道 4 | GPIO19 | gpiochip0:19 |
| 通道 5 | GPIO20 | gpiochip0:20 |
| 通道 6 | GPIO21 | gpiochip0:21 |
| 通道 7 | GPIO22 | gpiochip0:22 |
| 通道 8 | GPIO23 | gpiochip0:23 |

> **记下来**：用 `gpioinfo` 找到上述 GPIO 的实际 chip 和 line 编号。
> RK3588 的 GPIO 编号可能和树莓派不同，但 40pin 物理位置是一样的。

"""PWR.A153 B GPIO PWM 舵机驱动"""
import time
import os
from typing import List


class ServoDriver:
    """
    通过 PWR.A153 B 扩展板的 GPIO PWM 控制 SG90 舵机。

    使用 Linux sysfs PWM 接口（不需要 root 权限时用 lgpio）。
    """

    # 舵机参数
    PWM_FREQ = 50           # Hz
    PWM_PERIOD_NS = 20000000  # 20ms (单位: 纳秒)

    # 角度 ↔ 脉宽映射
    # SG90: 0°=0.5ms, 90°=1.5ms, 180°=2.5ms
    MIN_PULSE_NS = 500000    # 0.5ms
    MAX_PULSE_NS = 2500000   # 2.5ms

    def __init__(self, pwm_chip: int = 0, channels: List[int] = None):
        """
        Args:
            pwm_chip: PWM 控制器编号 (/sys/class/pwm/pwmchip{pwm_chip})
            channels: 8个舵机对应的 PWM 通道号，例如 [0,1,2,3,4,5,6,7]
        """
        self.pwm_chip = pwm_chip
        self.channels = channels or list(range(8))
        self._exported = set()

    def initialize(self) -> bool:
        """初始化 PWM，导出所有通道"""
        pwmchip_path = f"/sys/class/pwm/pwmchip{self.pwm_chip}"
        if not os.path.exists(pwmchip_path):
            print(f"[ServoDriver] {pwmchip_path} 不存在，尝试 lgpio 方式")

        for ch in self.channels:
            self._export_pwm(ch)
            self._set_period(ch)
        print(f"[ServoDriver] {len(self.channels)} 个通道初始化完成")
        return True

    def _export_pwm(self, channel: int):
        """导出 PWM 通道"""
        export_path = f"/sys/class/pwm/pwmchip{self.pwm_chip}/export"
        pwm_path = f"/sys/class/pwm/pwmchip{self.pwm_chip}/pwm{channel}"
        if not os.path.exists(pwm_path):
            try:
                with open(export_path, 'w') as f:
                    f.write(str(channel))
                time.sleep(0.1)
            except PermissionError:
                print(f"[ServoDriver] 导出 PWM{channel} 需要 root 权限")
                print("  执行: sudo chmod 666 /sys/class/pwm/pwmchip*/export")
        self._exported.add(channel)

    def _set_period(self, channel: int):
        """设置 PWM 周期为 20ms（50Hz）"""
        period_path = f"/sys/class/pwm/pwmchip{self.pwm_chip}/pwm{channel}/period"
        try:
            with open(period_path, 'w') as f:
                f.write(str(self.PWM_PERIOD_NS))
        except Exception as e:
            print(f"[ServoDriver] 设置周期失败 ch{channel}: {e}")

    def set_angle(self, channel: int, angle: float):
        """设置舵机角度 (0-180)"""
        angle = max(0, min(180, angle))
        # 线性映射: 0°→MIN_PULSE, 180°→MAX_PULSE
        pulse_ns = int(self.MIN_PULSE_NS +
                       (angle / 180.0) * (self.MAX_PULSE_NS - self.MIN_PULSE_NS))

        duty_path = f"/sys/class/pwm/pwmchip{self.pwm_chip}/pwm{channel}/duty_cycle"
        enable_path = f"/sys/class/pwm/pwmchip{self.pwm_chip}/pwm{channel}/enable"

        try:
            with open(enable_path, 'w') as f:
                f.write("1")
            with open(duty_path, 'w') as f:
                f.write(str(pulse_ns))
        except Exception as e:
            print(f"[ServoDriver] 设置角度失败 ch{channel}: {e}")

    def open_locker(self, channel: int):
        """开锁：舵机转 90°"""
        self.set_angle(channel, 90)
        time.sleep(0.8)
        self.stop(channel)  # 到位后停止 PWM，防止堵转

    def close_locker(self, channel: int):
        """闭锁：舵机转 0°"""
        self.set_angle(channel, 0)
        time.sleep(0.8)
        self.stop(channel)

    def stop(self, channel: int):
        """停止 PWM 信号"""
        enable_path = f"/sys/class/pwm/pwmchip{self.pwm_chip}/pwm{channel}/enable"
        try:
            with open(enable_path, 'w') as f:
                f.write("0")
        except:
            pass

    def shutdown(self):
        """释放所有 PWM 通道"""
        for ch in list(self._exported):
            self.stop(ch)
            unexport_path = f"/sys/class/pwm/pwmchip{self.pwm_chip}/unexport"
            try:
                with open(unexport_path, 'w') as f:
                    f.write(str(ch))
            except:
                pass
        print("[ServoDriver] 已释放所有通道")


# ── 舵机调试脚本 ──
if __name__ == "__main__":
    import sys

    driver = ServoDriver()
    driver.initialize()

    print("舵机调试工具")
    print("命令: o0=开1号锁  c0=关1号锁  q=退出")
    print(f"可用通道: 0~{len(driver.channels)-1}")

    while True:
        cmd = input("> ").strip()
        if cmd == 'q':
            break
        if cmd.startswith('o'):
            ch = int(cmd[1:])
            driver.open_locker(ch)
            print(f"  通道{ch} 开锁 (90°)")
        elif cmd.startswith('c'):
            ch = int(cmd[1:])
            driver.close_locker(ch)
            print(f"  通道{ch} 闭锁 (0°)")

    driver.shutdown()

```python
"""霍尔传感器 GPIO 读取 — 门状态检测"""
import lgpio
from typing import Dict, List


class DoorSensor:
    """
    霍尔传感器 + 磁铁 门状态检测。
    磁铁靠近 → 传感器输出低电平 → 门关闭
    磁铁远离 → 传感器输出高电平 → 门打开
    """

    # 8 个门对应的 GPIO 引脚（chip, line）
    # 根据实际接线修改
    DEFAULT_DOOR_PINS = [5, 6, 7, 8, 9, 10, 11, 12]

    def __init__(self, gpio_pins: List[int] = None):
        self.pins = gpio_pins or self.DEFAULT_DOOR_PINS
        self._chip = None

    def initialize(self) -> bool:
        """初始化 GPIO 为输入模式"""
        self._chip = lgpio.gpiochip_open(0)
        for pin in self.pins:
            # 设置为输入，上拉（传感器开漏输出）
            lgpio.gpio_claim_input(self._chip, pin, lgpio.SET_PULL_UP)
        print(f"[DoorSensor] {len(self.pins)} 个门传感器初始化完成")
        return True

    def get_status(self, locker_id: int) -> str:
        """
        读取指定柜门状态。
        返回: 'closed' | 'open' | 'unknown'
        """
        if locker_id < 1 or locker_id > len(self.pins):
            return 'unknown'
        pin = self.pins[locker_id - 1]
        level = lgpio.gpio_read(self._chip, pin)
        return 'closed' if level == 0 else 'open'

    def get_all_status(self) -> Dict[int, str]:
        """返回所有柜门状态"""
        return {i + 1: self.get_status(i + 1)
                for i in range(len(self.pins))}

    def wait_for_close(self, locker_id: int, timeout: float = 10.0):
        """等待柜门关闭（阻塞）"""
        import time
        t0 = time.time()
        while time.time() - t0 < timeout:
            if self.get_status(locker_id) == 'closed':
                return True
            time.sleep(0.1)
        return False

    def shutdown(self):
        if self._chip:
            lgpio.gpiochip_close(self._chip)
```

"""柜门状态 LED 控制"""
import lgpio
from typing import List


class LEDControl:
    """RK3588 GPIO 直驱 LED"""

    DEFAULT_LED_PINS = [24, 25, 26, 27, 28, 29, 30, 31]

    def __init__(self, pins: List[int] = None):
        self.pins = pins or self.DEFAULT_LED_PINS
        self._chip = None

    def initialize(self) -> bool:
        self._chip = lgpio.gpiochip_open(0)
        for pin in self.pins:
            lgpio.gpio_claim_output(self._chip, pin)
        return True

    def set(self, locker_id: int, state: str):
        """state: 'red'(高电平), 'green'(高电平), 'off'(低电平)"""
        # 简化：红色 GPIO 和绿色 GPIO 各占一半
        # 实际使用时按硬件接线调整
        if locker_id < 1 or locker_id > len(self.pins):
            return
        pin = self.pins[locker_id - 1]
        if state == 'off':
            lgpio.gpio_write(self._chip, pin, 0)
        else:
            lgpio.gpio_write(self._chip, pin, 1)

    def shutdown(self):
        for pin in self.pins:
            lgpio.gpio_write(self._chip, pin, 0)
        if self._chip:
            lgpio.gpiochip_close(self._chip)

### 5.2 `src/auth/nfc_reader.py`

```python
"""PN532 NFC 读卡器 — 读取校园卡 UID"""
import serial
import time
from typing import Optional


class NFCReader:
    """PN532 NFC 模块，UART 模式"""

    def __init__(self, port: str = "/dev/ttyS5", baudrate: int = 115200):
        self.port = port
        self.baudrate = baudrate
        self._serial = None

    def initialize(self) -> bool:
        try:
            self._serial = serial.Serial(self.port, self.baudrate, timeout=0.5)
            # 唤醒 PN532
            self._wake_up()
            # 获取固件版本（验证通信正常）
            version = self._get_firmware_version()
            if version:
                print(f"[NFCReader] PN532 固件版本: {version}")
                return True
            return False
        except Exception as e:
            print(f"[NFCReader] 初始化失败: {e}")
            return False

    def _wake_up(self):
        """唤醒 PN532"""
        self._serial.write(b'\x55\x55\x00\x00\x00\x00\x00\x00\x00\x00')
        time.sleep(0.1)

    def _send_command(self, command: bytes) -> Optional[bytes]:
        """发送命令并接收响应"""
        if not self._serial:
            return None
        try:
            self._serial.write(command)
            time.sleep(0.05)
            # 读取 ACK
            ack = self._serial.read(6)
            if len(ack) < 6:
                return None
            # 读取响应
            response = self._serial.read(32)
            return response
        except:
            return None

    def _get_firmware_version(self) -> Optional[str]:
        """获取固件版本"""
        cmd = bytes([0x00, 0x00, 0xFF, 0x02, 0xFE, 0xD4, 0x02, 0x2A, 0x00])
        resp = self._send_command(cmd)
        if resp and len(resp) >= 7:
            return f"{resp[5]}.{resp[6]}"
        return None

    def read_card_uid(self, timeout: float = 10.0) -> Optional[str]:
        """
        阻塞等待刷卡，返回卡片 UID（16进制字符串）。
        超时返回 None。
        """
        # PN532 InListPassiveTarget 命令
        cmd = bytes([
            0x00, 0x00, 0xFF,
            0x04,  # 数据长度
            0xFC,  # 校验
            0xD4, 0x4A,  # InListPassiveTarget
            0x01, 0x00,  # MaxTg=1, BrTy=106kbps
            0x00   # 校验
        ])

        t0 = time.time()
        while time.time() - t0 < timeout:
            resp = self._send_command(cmd)
            if resp and len(resp) >= 10 and resp[7] > 0:
                # 提取 UID
                uid_len = resp[12]
                uid_bytes = resp[13:13 + uid_len]
                return uid_bytes.hex().upper()
            time.sleep(0.3)

        return None

    def shutdown(self):
        if self._serial:
            self._serial.close()
```

```python
"""人脸检测与识别 — OpenCV + Dlib"""
import os
import cv2
import numpy as np
from typing import Optional, List
from src.interfaces import AuthResult


class FaceRecognizer:
    """人脸检测 + 特征提取 + 比对"""

    def __init__(self, camera_device: int = 0):
        self.camera_device = camera_device
        self._cap = None
        self._detector = None

    def initialize(self) -> bool:
        """初始化摄像头和 Dlib 模型"""
        try:
            import dlib
            self._detector = dlib.get_frontal_face_detector()
            self._predictor = dlib.shape_predictor(
                "models/shape_predictor_68_face_landmarks.dat"
            )
            self._recognizer = dlib.face_recognition_model_v1(
                "models/dlib_face_recognition_resnet_model_v1.dat"
            )
            self._cap = cv2.VideoCapture(self.camera_device)
            if not self._cap.isOpened():
                print("[FaceRec] 摄像头打开失败")
                return False
            print("[FaceRec] 人脸识别模块初始化完成")
            return True
        except ImportError:
            print("[FaceRec] dlib 未安装，使用简化模式")
            self._cap = cv2.VideoCapture(self.camera_device)
            return True
        except Exception as e:
            print(f"[FaceRec] 初始化失败: {e}")
            return False

    def _download_dlib_models(self):
        """下载 Dlib 预训练模型（首次运行）"""
        import urllib.request
        os.makedirs("models", exist_ok=True)

        models = {
            "shape_predictor_68_face_landmarks.dat":
                "http://dlib.net/files/shape_predictor_68_face_landmarks.dat.bz2",
            "dlib_face_recognition_resnet_model_v1.dat":
                "http://dlib.net/files/dlib_face_recognition_resnet_model_v1.dat.bz2",
        }
        for fname, url in models.items():
            fpath = f"models/{fname}"
            if not os.path.exists(fpath):
                print(f"[FaceRec] 下载模型: {fname} (~100MB, 仅首次)")
                # 实际下载逻辑
                # urllib.request.urlretrieve(url, fpath + ".bz2")
                # bz2 解压...

    def capture_face(self) -> Optional[np.ndarray]:
        """从摄像头捕获一帧包含人脸的图像"""
        if not self._cap:
            return None
        for _ in range(10):  # 尝试 10 帧
            ret, frame = self._cap.read()
            if not ret:
                continue
            if self._detector:
                faces = self._detector(frame, 1)
                if len(faces) > 0:
                    return frame
            else:
                # 简化模式：用 OpenCV Haar Cascade
                gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
                cascade = cv2.CascadeClassifier(
                    cv2.data.haarcascades + 'haarcascade_frontalface_default.xml'
                )
                faces = cascade.detectMultiScale(gray, 1.1, 4)
                if len(faces) > 0:
                    return frame
        return None

    def extract_features(self, image: np.ndarray) -> Optional[List[float]]:
        """提取 128 维人脸特征向量"""
        if not self._detector:
            # 简化模式：返回假特征
            return list(np.random.randn(128))

        faces = self._detector(image, 1)
        if len(faces) == 0:
            return None

        shape = self._predictor(image, faces[0])
        features = self._recognizer.compute_face_descriptor(image, shape)
        return list(features)

    def compare(self, feat1: List[float], feat2: List[float],
                threshold: float = 0.6) -> bool:
        """欧氏距离比对，< threshold 视为同一个人"""
        a = np.array(feat1)
        b = np.array(feat2)
        distance = np.linalg.norm(a - b)
        return distance < threshold

    def verify(self, known_features: dict) -> AuthResult:
        """
        采集人脸并与已知特征比对。

        Args:
            known_features: {user_id: [128维特征], ...}
                Person C 通过 DataManager.get_registered_faces() 获取
        """
        image = self.capture_face()
        if image is None:
            return AuthResult(success=False, error_msg="未检测到人脸")

        features = self.extract_features(image)
        if features is None:
            return AuthResult(success=False, error_msg="无法提取面部特征")

        # 与所有注册用户比对
        for user_id, known_feat in known_features.items():
            if self.compare(features, known_feat):
                return AuthResult(
                    success=True, user_id=user_id,
                    permission_level=1,  # Person C 查数据库填
                    method="face"
                )

        return AuthResult(success=False, error_msg="人脸未注册")

    def register(self) -> Optional[List[float]]:
        """注册新人脸：采集3次取平均"""
        features_list = []
        for i in range(3):
            print(f"[FaceRec] 第{i+1}次采集，请正视摄像头...")
            image = self.capture_face()
            if image is None:
                continue
            feat = self.extract_features(image)
            if feat:
                features_list.append(feat)

        if len(features_list) < 2:
            return None

        # 取平均
        avg = np.mean(features_list, axis=0).tolist()
        print(f"[FaceRec] 人脸注册完成（{len(features_list)}次采样）")
        return avg

    def shutdown(self):
        if self._cap:
            self._cap.release()
```

## Step 7: HardwareController 实现（1天）

将所有模块整合到 `src/hardware/hw_controller.py`，实现 `src/interfaces.py` 中定义的 `HardwareController` 接口：

```python
"""HardwareController 完整实现"""
from typing import Dict, Optional, List
from src.interfaces import HardwareController, AuthResult

# 导入各模块
from src.hardware.servo_driver import ServoDriver
from src.hardware.door_sensor import DoorSensor
from src.hardware.led_control import LEDControl
from src.auth.nfc_reader import NFCReader
from src.auth.face_recognizer import FaceRecognizer


class HardwareControllerImpl(HardwareController):

    def __init__(self, locker_count: int = 8):
        self.locker_count = locker_count
        self.servo = ServoDriver()
        self.sensors = DoorSensor()
        self.leds = LEDControl()
        self.nfc = NFCReader()
        self.face = FaceRecognizer()

    def initialize(self) -> bool:
        ok = True
        ok &= self.servo.initialize()
        ok &= self.sensors.initialize()
        ok &= self.leds.initialize()
        ok &= self.nfc.initialize()
        ok &= self.face.initialize()
        print(f"[HWController] 初始化{'成功' if ok else '部分失败'}")
        return ok

    # ── 柜门控制 ──
    def open_locker(self, locker_id: int) -> bool:
        if locker_id < 1 or locker_id > self.locker_count:
            return False
        self.leds.set(locker_id, 'red')
        self.servo.open_locker(locker_id - 1)
        return True

    def close_locker(self, locker_id: int) -> bool:
        if locker_id < 1 or locker_id > self.locker_count:
            return False
        self.leds.set(locker_id, 'green')
        self.servo.close_locker(locker_id - 1)
        return True

    def get_all_door_status(self) -> Dict[int, str]:
        return self.sensors.get_all_status()

    # ── 身份验证 ──
    def read_nfc_card(self, timeout: float = 10.0) -> Optional[str]:
        return self.nfc.read_card_uid(timeout)

    def verify_face(self, known_features: Dict = None) -> AuthResult:
        return self.face.verify(known_features or {})

    def register_face(self) -> Optional[List[float]]:
        return self.face.register()

    # ── 条形码 ──
    def scan_barcode(self, timeout: float = 15.0) -> Optional[str]:
        # 简化实现：USB HID 键盘模拟
        # Person C 也可以用 evdev 直接实现
        return input(f"[Barcode] 请输入条形码 (模拟): ")

    # ── LED ──
    def set_led(self, locker_id: int, state: str):
        self.leds.set(locker_id, state)

    def shutdown(self):
        self.servo.shutdown()
        self.sensors.shutdown()
        self.leds.shutdown()
        self.nfc.shutdown()
        self.face.shutdown()
```
