```python
"""SQLite 数据库连接管理"""
import sqlite3
import os


class DBManager:
    """数据库连接管理和建表"""

    def __init__(self, db_path: str = "data/components.db"):
        self.db_path = db_path
        self._conn = None

    def initialize(self) -> bool:
        os.makedirs(os.path.dirname(self.db_path), exist_ok=True)
        self._conn = sqlite3.connect(self.db_path)
        self._conn.row_factory = sqlite3.Row
        self._conn.execute("PRAGMA journal_mode=WAL")
        self._conn.execute("PRAGMA foreign_keys=ON")
        self._create_tables()
        print(f"[DB] 数据库连接成功: {self.db_path}")
        return True

    def _create_tables(self):
        """建表"""
        self._conn.executescript("""
            CREATE TABLE IF NOT EXISTS components (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT NOT NULL,
                category TEXT,
                voltage REAL,
                current REAL,
                package TEXT DEFAULT '',
                cabinet_id INTEGER,
                keywords TEXT DEFAULT '',
                stock INTEGER DEFAULT 0,
                barcode TEXT DEFAULT ''
            );

            CREATE TABLE IF NOT EXISTS users (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                name TEXT NOT NULL,
                card_id TEXT UNIQUE,
                face_features TEXT,
                permission_level INTEGER DEFAULT 1,
                is_admin INTEGER DEFAULT 0,
                admin_password TEXT,
                account_status INTEGER DEFAULT 1
            );

            CREATE TABLE IF NOT EXISTS records (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                user_id INTEGER,
                component_id INTEGER,
                cabinet_id INTEGER,
                borrow_time TEXT DEFAULT (datetime('now','localtime')),
                return_time TEXT,
                borrow_mode TEXT DEFAULT 'ai_assist',
                status TEXT DEFAULT 'borrowed',
                FOREIGN KEY (user_id) REFERENCES users(id),
                FOREIGN KEY (component_id) REFERENCES components(id)
            );
        """)
        self._conn.commit()

    @property
    def conn(self):
        return self._conn

    def shutdown(self):
        if self._conn:
            self._conn.close()
```

```python
"""器件数据仓库 — CRUD + 搜索"""
import json
from typing import List, Optional
from src.database.db_manager import DBManager
from src.interfaces import Component


class ComponentRepo:
    def __init__(self, db: DBManager):
        self.db = db

    def init_from_json(self, json_path: str):
        """从 JSON 文件导入初始数据"""
        with open(json_path, 'r', encoding='utf-8') as f:
            data = json.load(f)

        for comp in data.get("components", []):
            self.db.conn.execute("""
                INSERT OR REPLACE INTO components
                (id, name, category, voltage, current, package,
                 cabinet_id, keywords, stock)
                VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
            """, (
                comp["id"], comp["name"], comp["category"],
                comp.get("voltage"), comp.get("current"),
                comp.get("package", ""), comp["cabinet_id"],
                comp.get("keywords", ""), comp.get("stock", 0)
            ))
        self.db.conn.commit()
        print(f"[Repo] 导入 {len(data.get('components', []))} 个器件")

    def search(self, voltage: float = None, category: str = None,
               keyword: str = None, limit: int = 5) -> List[Component]:
        """多条件搜索"""
        query = "SELECT * FROM components WHERE 1=1"
        params = []

        if voltage is not None:
            query += " AND voltage BETWEEN ? AND ?"
            params.extend([voltage - 0.5, voltage + 0.5])

        if category:
            query += " AND category LIKE ?"
            params.append(f"%{category}%")

        if keyword:
            query += " AND (name LIKE ? OR keywords LIKE ?)"
            params.extend([f"%{keyword}%", f"%{keyword}%"])

        query += " ORDER BY stock DESC LIMIT ?"
        params.append(limit)

        rows = self.db.conn.execute(query, params).fetchall()
        return [Component(
            id=r["id"], name=r["name"], category=r["category"] or "",
            voltage=r["voltage"], current=r["current"],
            package=r["package"] or "", cabinet_id=r["cabinet_id"] or 0,
            keywords=r["keywords"] or "", stock=r["stock"] or 0,
        ) for r in rows]

    def get_by_id(self, comp_id: int) -> Optional[Component]:
        r = self.db.conn.execute(
            "SELECT * FROM components WHERE id=?", (comp_id,)
        ).fetchone()
        if not r:
            return None
        return Component(
            id=r["id"], name=r["name"], category=r["category"] or "",
            voltage=r["voltage"], current=r["current"],
            package=r["package"] or "", cabinet_id=r["cabinet_id"] or 0,
            keywords=r["keywords"] or "", stock=r["stock"] or 0,
        )

    def get_by_cabinet(self, cabinet_id: int) -> Optional[Component]:
        r = self.db.conn.execute(
            "SELECT * FROM components WHERE cabinet_id=?", (cabinet_id,)
        ).fetchone()
        if not r:
            return None
        return self.get_by_id(r["id"])

    def get_by_barcode(self, barcode: str) -> Optional[Component]:
        r = self.db.conn.execute(
            "SELECT * FROM components WHERE barcode=?", (barcode,)
        ).fetchone()
        if not r:
            return None
        return self.get_by_id(r["id"])

    def get_all_for_llm(self) -> str:
        """生成供 LLM 使用的器件目录文本"""
        rows = self.db.conn.execute("SELECT * FROM components ORDER BY id").fetchall()
        lines = []
        for r in rows:
            lines.append(
                f"{r['id']}. {r['name']} | {r['category'] or ''} | "
                f"电压:{r['voltage']}V 电流:{r['current']}A | "
                f"柜门{r['cabinet_id']} | 关键词:{r['keywords'] or ''}"
            )
        return "\n".join(lines)

    def update_stock(self, comp_id: int, delta: int):
        """更新库存（借用 -1，归还 +1）"""
        self.db.conn.execute(
            "UPDATE components SET stock = stock + ? WHERE id = ?",
            (delta, comp_id)
        )
        self.db.conn.commit()
```

```python
"""USB 条形码扫描器监听 — evdev 方式"""
import evdev
import threading
from typing import Callable


class BarcodeListener:
    """
    USB HID 条形码扫描器（键盘模拟模式）监听。
    扫描器输入等同于快速键盘敲击 + 回车，通过 evdev 捕获。
    """

    def __init__(self, callback: Callable[[str], None]):
        self.callback = callback
        self._buffer = ""
        self._running = False
        self._last_event = 0

    def start(self):
        """启动后台监听线程"""
        self._running = True
        threading.Thread(target=self._listen, daemon=True).start()
        print("[Barcode] 监听已启动")

    def _listen(self):
        import time
        # 查找条形码扫描器设备（通常名称含 "Bar" 或 "Scanner"）
        devices = [evdev.InputDevice(path) for path in evdev.list_devices()]
        scanner = None
        for dev in devices:
            if any(kw in dev.name.lower() for kw in ['bar', 'scan', 'barcode', 'honeywell']):
                scanner = dev
                break

        # 如果没找到，监听所有键盘设备
        if not scanner:
            for dev in devices:
                if 'key' in dev.name.lower() or 'input' in dev.name.lower():
                    scanner = dev
                    break

        if not scanner:
            print("[Barcode] 未找到扫描器，请在管理员界面手动输入")
            return

        print(f"[Barcode] 使用设备: {scanner.name}")

        for event in scanner.read_loop():
            if not self._running:
                break
            if event.type == evdev.ecodes.EV_KEY and event.value == 1:
                key = evdev.categorize(event).keycode
                if key == 'KEY_ENTER':
                    if self._buffer:
                        self.callback(self._buffer)
                        self._buffer = ""
                elif len(key) == 1:  # 单字符键
                    self._buffer += key
```

```python
                self.ai.update_catalog(catalog)

            # 4. 语音
            info("初始化语音模块...")
            self.voice_synthesizer = VoiceSynthesizer()
            self.voice_recognizer = VoiceRecognizer()
            self.interaction = InteractionManager(
                self.ai, self.voice_recognizer, self.voice_synthesizer
            )

            # 5. GUI
            info("启动图形界面...")
            self.gui = MainWindow(system=self)
            self.gui.show()

            info("系统初始化完成")
            return True

        except Exception as e:
            error(f"初始化失败: {e}")
            return False

    def handle_borrow_request(self, query_text: str):
        """
        借用流程：语音查询 → AI 匹配 → 展示结果 → 开锁

        这是整个系统最核心的业务逻辑。
        Person C 负责编写，调用 A 和 B 的接口。
        """
        info(f"收到借用请求: {query_text}")

        # Step 1: AI 语义匹配（调用 Person A）
        if self.ai:
            result = self.ai.process_query(query_text)
        else:
            info("AI 引擎未加载，无法处理")
            return

        # Step 2: 检查匹配结果
        if result.confidence < 0.3:
            self.speak(result.response_text)
            return

        # Step 3: 身份验证（调用 Person B）
        auth = self._authenticate_user()
        if not auth.success:
            self.speak("身份验证失败，请重试")
            return

        # Step 4: 开锁（调用 Person B）
        cabinet_id = result.cabinet_id
        if cabinet_id and self.hw:
            success = self.hw.open_locker(cabinet_id)
            if success:
                # Step 5: 记录（自己的数据库）
                self.data.create_borrow_record(
                    user_id=auth.user_id,
                    component_id=result.component_id,
                    cabinet_id=cabinet_id,
                )
                self.speak(f"{cabinet_id}号柜门已打开，请取用{result.component_name}")
            else:
                self.speak("柜门打开失败，请联系管理员")
        else:
            self.speak(result.response_text)

    def _authenticate_user(self):
        """身份验证流程 — 人脸 或 NFC 任选其一"""
        from src.interfaces import AuthResult

        # 先尝试人脸识别（调用 Person B）
        if self.hw:
            faces = self.data.get_registered_faces()
            result = self.hw.verify_face(faces)
            if result.success:
                return result

        # 人脸失败，尝试 NFC（调用 Person B）
        self.speak("请刷卡")
        if self.hw:
            card_id = self.hw.read_nfc_card(timeout=15)
            if card_id:
                user = self.data.get_user_by_card(card_id)
                if user:
                    return AuthResult(
                        success=True, user_id=user.id,
                        user_name=user.name,
                        permission_level=user.permission_level,
                        method="nfc"
                    )

        return AuthResult(success=False, error_msg="验证失败")

    def speak(self, text: str):
        """语音播报"""
        info(f"播报: {text}")
        if self.voice_synthesizer:
            self.voice_synthesizer.speak_async(text)

    def run(self):
        """启动系统"""
        self.running = True
        if self.interaction:
            self.interaction.start()
        # GUI 事件循环（阻塞）
        from PyQt5.QtWidgets import QApplication
        sys.exit(QApplication.instance().exec_())

    def stop(self):
        """停止系统"""
        self.running = False
        if self.ai:
            self.ai.shutdown()
        if self.hw:
            self.hw.shutdown()
        if self.data:
            self.data.shutdown()
```
